#include<bits/stdc++.h>
using namespace std;
int main(){
        string a, b, c;
    cin >> a >> b >> c;

    int freq[26] = {0};

    for (char ch : a) freq[ch - 'A']++;
    for (char ch : b) freq[ch - 'A']++;

    for (char ch : c) freq[ch - 'A']--;

    for (int i = 0; i < 26; i++) {
        if (freq[i] != 0) {
            cout << "NO\n";
            return 0;
        }
    }

    cout << "YES\n";
    return 0;
}

#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;

    int police = 0;
    int untreated = 0;

    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;

        if (x == -1) {
            if (police > 0) {
                police--; 
            } else {
                untreated++; 
            }
        } else {
            police += x; 
        }
    }

    cout << untreated << endl;

    return 0;
}
