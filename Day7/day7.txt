#include <bits/stdc++.h>
using namespace std;
int main() {
    int a[4];
    for (int i = 0; i < 4; i++) cin >> a[i];
    int cnt = 0;
    for (int i = 0; i < 4; i++) {
        bool seen = false;
        for (int j = 0; j < i; j++) {
            if (a[i] == a[j]) {
                seen = true;
                break;
            }
        }
        if (!seen) cnt++;
    }
    cout << 4 - cnt << endl;
}

int main() {
    string a, b;
    cin >> a >> b;
    string ans = "";
    for (int i = 0; i < a.size(); i++) {
        if (a[i] == b[i])
            ans += '0';
        else
            ans += '1';
    }
    cout << ans << endl;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    cin >> n;
    string res = "";
    for (int i = 1; i <= n; i++) {
        if (i % 2 == 1)
            res += "I hate";
        else
            res += "I love";

        if (i != n)
            res += " that ";
        else
            res += " it";
    }
    cout << res << endl;
    return 0;
}

using namespace std;
int main() {
    int t;
    cin >> t;
    while (t--) {
        long long a, b;
        cin >> a >> b;

        long long rem = a % b;

        if (rem == 0)
            cout << 0 << endl;
        else
            cout << b - rem << endl;
    }
    return 0;
}

int main() {
    int n;
    cin >> n;
    string s;
    cin >> s;
    set<char> st;
    for (char c : s) {
        st.insert(tolower(c));
    }
    if (st.size() == 26)
        cout << "YES" << endl;
    else
        cout << "NO" << endl;
    return 0;
}
